## **üõ†Ô∏è Automation Scripts**

### **Google Drive Upload Script**
```python
from pydrive.auth import GoogleAuth
from pydrive.drive import GoogleDrive
import os

gauth = GoogleAuth()
gauth.LoadCredentialsFile(os.getenv("SERVICE_ACCOUNT_KEY_PATH"))
drive = GoogleDrive(gauth)

folder_id = os.getenv("DRIVE_FOLDER_ID")

for file_name in os.listdir("."):
    if file_name.endswith(".csv"):
        file = drive.CreateFile({"parents": [{"id": folder_id}], "title": file_name})
        file.SetContentFile(file_name)
        file.Upload()
        print(f"Uploaded: {file_name}")
```

### **Automated Email Notifications**
```python
import smtplib
from email.mime.text import MIMEText
import os

EMAIL = os.getenv("GMAIL_CREDENTIALS_PATH")
PASSWORD = os.getenv("GMAIL_TOKEN_PATH")

def send_email(subject, body):
    msg = MIMEText(body)
    msg['Subject'] = subject
    msg['From'] = EMAIL
    msg['To'] = "samq@damg.com"

    with smtplib.SMTP('smtp.gmail.com', 587) as server:
        server.starttls()
        server.login(EMAIL, PASSWORD)
        server.sendmail(EMAIL, "samq@damg.com", msg.as_string())
```
